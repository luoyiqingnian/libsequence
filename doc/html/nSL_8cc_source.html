59  Mechanics of the nSL statistic
60 
61  RV = nSL,iHS, as defined in doi:10.1093/molbev/msu077
62  */
63  std::array<double, 4>
64  __nSLdetails(const std::size_t &core, const Sequence::SimData &d,
65  // const vector<size_t> &coretype,
66  const std::unordered_map<double, double> &gmap)
67  {
68  auto csize = d.size();
69  // This tracks s,s2 for ancestral and derived
70  // mutation, resp:
71  std::array<double, 4> rv = { 0., 0., 0., 0. };
72  // number of comparisons for ancestral and
73  // derived, resp:
74  std::array<unsigned, 2> nc = { 0u, 0u };
75  for (size_t i = 0; i < csize; ++i)
76  {
77  auto start = d[i].cbegin();
78  auto bi
79  = start
80  + static_cast<decltype(start)::difference_type>(core);
81  size_t iIsDer = (*bi == '1');
82  for (size_t j = i + 1; j < csize; ++j)
83  {
84  auto bj
85  = d[j].cbegin()
86  + static_cast<decltype(start)::difference_type>(
87  core);
88  size_t jIsDer = (*bj == '1');
89  if (iIsDer == jIsDer)
90  {
91  auto eri = d[i].crend();
92  auto ei = d[i].cend();
93  auto right = mismatch(bi, ei, bj);
94  string::const_reverse_iterator ri1(bi),
95  ri2(bj);
96  auto left = mismatch(ri1, eri, ri2);
97  if (left.first != eri && right.first != ei)
98  {
99  rv[2 * iIsDer] += static_cast<double>(
100  distance(left.first.base(),
101  right.first)
102  + 1);
103  rv[2 * iIsDer + 1] += update_s2(
104  start, left.first.base(),
105  right.first, d, gmap);
106  nc[iIsDer]++;
107  }
108  }
109  }
110  }
111  rv[0] /= static_cast<double>(nc[0]);
112  rv[1] /= static_cast<double>(nc[0]);
113  rv[2] /= static_cast<double>(nc[1]);
114  rv[3] /= static_cast<double>(nc[1]);
115  return rv;
116  }

141  vector<tuple<double, double, uint32_t>>
142  nSL_t(const SimData &d, const std::vector<size_t> &core_snps,
143  const std::unordered_map<double, double> &gmap)
144  {
145  vector<tuple<double, double, uint32_t>> rv(core_snps.size());
146  using offset_type = SimData::const_site_iterator::difference_type;
147  tbb::parallel_for(
148  tbb::blocked_range<std::size_t>(0, rv.size()),
149  [&rv, &d, &core_snps,
150  &gmap](const tbb::blocked_range<std::size_t> &r) {
151  for (std::size_t i = r.begin(); i < r.end(); ++i)
152  {
153  auto temp = __nSLdetails(core_snps[i], d, gmap);
154  offset_type offset = static_cast<offset_type>(i);
155  uint32_t dcount = static_cast<uint32_t>(
156  count((d.sbegin() + offset)->second.begin(),
157  (d.sbegin() + offset)->second.end(), '1'));
158  rv[i]
159  = make_tuple(log(temp[0]) - log(temp[2]),
160  log(temp[1]) - log(temp[3]), dcount);
161  }
162  });
163  return rv;
164  }

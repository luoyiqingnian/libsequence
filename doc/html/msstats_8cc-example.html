#include <iostream>
#include <vector>
#include <Sequence/SimParams.hpp>
#include <Sequence/SimData.hpp>
#include <Sequence/PolySIM.hpp>
#include <cstdio>
using namespace std;
using namespace Sequence;
int main(int argc, char *argv[])
{
SimParams p;
cin >> p;
SimData d;
std::ios_base::sync_with_stdio(true);
int rv;
while( (rv = d.fromfile(stdin)) != EOF )
{
PolySIM P(&d);
cout <<P.NumPoly() << '\t'
<< P.ThetaW() << '\t'
<< P.ThetaPi() << '\t'
<< P.ThetaH() << '\t'
<< P.TajimasD() << '\t'
<< P.FuLiD() << '\t'
<< P.FuLiF() << '\t'
<< P.FuLiDStar() << '\t'
<< P.FuLiFStar() << endl;
}
}
